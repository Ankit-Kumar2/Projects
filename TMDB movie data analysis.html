#Create a user defined function to split the value of Revenue column in to bins to analyze properly
def create_bins(df,field):
    min_val = df[field].min()
    _25_val = df[field].describe()[4]
    median = df[field].describe()[5]
    _75_val = df[field].describe()[6]
    max_val = df[field].max()
    bin_edges = [ min_val, _25_val, median, _75_val, max_val]
    bin_names = [ 'Low','Med' ,'Moderately High' ,'High' ] 
    name = '{}_bin'.format(field)
    df[name]=pd.cut(df[field], bin_edges, labels=bin_names)
    return df
